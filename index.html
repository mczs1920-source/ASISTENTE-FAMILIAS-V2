function doGet(e) { var codigo = e.parameter.codigo; var hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Hoja 1'); var datos = hoja.getDataRange().getValues(); var cabeceras = datos[0]; var primeracelda = datos[1][0]; var tipo = typeof primeracelda; for (var i = 1; i < datos.length; i++) { if (datos[i][0].toString().trim() === codigo.toString().trim()) { var residente = {}; for (var j = 0; j < cabeceras.length; j++) { residente[cabeceras[j]] = datos[i][j].toString(); } return ContentService .createTextOutput(JSON.stringify({ok: true, data: residente})) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ok: false, primeracelda: primeracelda, tipo: tipo, buscado: codigo})) .setMimeType(ContentService.MimeType.JSON); }
